#include <iostream.h>
#include <conio.h>

class Box
{
    int length;

public:
    // Constructor
    Box(int l)
    {
        this->length = l;   // using this pointer
        cout << "\nConstructor called";
    }

    void setLength(int l)
    {
        this->length = l;   // using this pointer
    }

    void show()
    {
        cout << "\nLength = " << length;
    }

    // Destructor
    ~Box()
    {
        cout << "\nDestructor called";
    }
};

void main()
{
    clrscr();

    // Dynamic object creation using pointer
    Box *b = new Box(10);

    b->show();
    b->setLength(20);
    b->show();

    // Free memory
    delete b;

    getch();
}
